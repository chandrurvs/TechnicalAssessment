CREATE SCHEMA VEHICLE;

DROP TABLE IF EXISTS VEHICLE.ref_vehicle_types;
CREATE TABLE VEHICLE.ref_vehicle_types (
  REF_VEHICLE_NO int(11) NOT NULL AUTO_INCREMENT,
  VEHICLE_TYPE_ID varchar(3) NOT NULL,
  VEHICLE_DESC varchar(45) NOT NULL,
   PRIMARY KEY (VEHICLE_TYPE_ID),
  UNIQUE KEY VEHICLE_NO_UNIQUE (REF_VEHICLE_NO)
);

DROP TABLE IF EXISTS VEHICLE.vehicle_details;

CREATE TABLE VEHICLE.vehicle_details (
  VEHICLE_DETAILS_ID int(11) NOT NULL AUTO_INCREMENT,
  VEHICLE_TYPE_ID varchar(3) NOT NULL,
  VEHICLE_NAME varchar(45) NOT NULL,
  MAKE varchar(45) DEFAULT NULL,
  MODEL varchar(45) DEFAULT NULL,
  Model_Year varchar(4) DEFAULT NULL,
  PRICE double DEFAULT NULL,
  CREATED_DATE TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  UPDATED_DATE TIMESTAMP,
  PRIMARY KEY (VEHICLE_DETAILS_ID),
  UNIQUE KEY VEHICLE_DETAILS_ID_UNIQUE (VEHICLE_DETAILS_ID),
  CONSTRAINT FK_Vechicle_Type_id FOREIGN KEY (VEHICLE_TYPE_ID) REFERENCES ref_vehicle_types (VEHICLE_TYPE_ID) ON DELETE NO ACTION ON UPDATE NO ACTION
);
